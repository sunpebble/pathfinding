```typescript
// Test script for Tongcheng crawler verification
import { crawlTongcheng } from './src/lib/crawlers/tongcheng.js';

async function test() {
  console.log('Starting Tongcheng crawler test...');

  const results = await crawlTongcheng('杭州', { maxPages: 1 });

  console.log(`\n=== RESULTS SUMMARY ===`);
  console.log(`Total guides extracted: ${results.length}`);

  // Analyze each result
  for (const result of results) {
    console.log(`\n--- ${result.title} ---`);
    console.log(`Content length: ${result.content?.length || 0} chars`);
    console.log(`Images: ${result.imageUrls?.length || 0}`);
    console.log(`Author: ${result.authorName}`);
    console.log(`Published: ${result.publishedAt || 'not found'}`);
    console.log(`Views: ${result.viewsCount || 0}`);
    console.log(`Quality score: ${result.qualityScore}`);

    // Check for placeholder content
    if (result.content?.includes('旅游攻略') && result.content.length < 100) {
      console.log('⚠️ WARNING: Appears to be placeholder content!');
    }
  }

  // Field extraction rates
  const stats = {
    totalGuides: results.length,
    withContent: results.filter((r) => r.content && r.content.length > 100)
      .length,
    withImages: results.filter((r) => r.imageUrls && r.imageUrls.length > 0)
      .length,
    withAuthor: results.filter(
      (r) => r.authorName && r.authorName !== '同程用户'
    ).length,
    withDate: results.filter((r) => r.publishedAt).length,
    withViews: results.filter((r) => r.viewsCount && r.viewsCount > 0).length,
    varyingQuality: new Set(results.map((r) => r.qualityScore)).size > 1,
  };

  console.log(`\n=== EXTRACTION RATES ===`);
  console.log(
    `Content (>100 chars): ${stats.withContent}/${stats.totalGuides}`
  );
  console.log(`Images: ${stats.withImages}/${stats.totalGuides}`);
  console.log(`Author (not default): ${stats.withAuthor}/${stats.totalGuides}`);
  console.log(`Publish date: ${stats.withDate}/${stats.totalGuides}`);
  console.log(`Views: ${stats.withViews}/${stats.totalGuides}`);
  console.log(`Quality varies: ${stats.varyingQuality}`);
}

test().catch(console.error);
```
